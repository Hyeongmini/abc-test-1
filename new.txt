#include <bits/stdc++.h>

using namespace std;

using ll = long long;
using pii = pair<int, int>;
using pll = pair<ll, ll>;

int N, H[101010];
set<pair<int, int>> M;
vector<pii> V, R;
set<pair<int, int>>::iterator it;
int e;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> N;
    for (int i = 1; i <= N; i++) {
        int l, r, h;
        cin >> l >> h >> r;
        V.emplace_back(l, i);
        V.emplace_back(r, -i);
        H[i] = h;
        e = max(e, r);
    }

    M.insert({0, 2e9});
    sort(V.begin(), V.end());

    for (int i = 0; i < V.size(); i++) {
        auto [x, t] = V[i];
        if (t > 0) {
            M.insert({H[t], t});
        } else if (t < 0) {
            t = -t;
            M.erase(M.find({H[t], t}));
        }
        if (i < V.size() - 1 && x != V[i+1].first) {
            int mh = 0;
            it = prev(M.end());
            mh = it->first;
            if (R.empty() || R.back().second != mh)
                R.emplace_back(x, mh);
        }
    }

    R.emplace_back(e, 0);
    for (auto [x, h]: R)
        cout << x << ' ' << h << ' ';
    

    return 0;
}
